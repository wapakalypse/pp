# ВНИМАНИЕ: Это временное решение для Node.js 14.x
RUN apt-get update && apt-get install -y curl gnupg2 && \
    # Создаем директорию для ключей
    mkdir -p /etc/apt/keyrings && \
    # Скачиваем и добавляем старый ключ (может не работать вечно)
    curl -fsSL https://deb.nodesource.com/gpgkey/nodesource.gpg.key | gpg --dearmor -o /etc/apt/keyrings/nodesource.gpg || true && \
    # Создаем запись репозитория для Node.js 14.x, указывая на использование ключа
    echo "deb [signed-by=/etc/apt/keyrings/nodesource.gpg] https://deb.nodesource.com/node_14.x nodistro main" | tee /etc/apt/sources.list.d/nodesource.list && \
    # Обновляем и устанавливаем
    apt-get update && apt-get install -y nodejs
